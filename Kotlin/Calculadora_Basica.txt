fun main() {
    
    //chama a função calcular() passando 3 argumentos.
    //1º o código da operação que será executada.
    //2º O primeiro número da expressão
    //3º O segundo número da expressão
    println(calcular(3,5.1,5.1))
}

fun calcular(oper:Int, priNum:Double, segNum:Double):String{
    var result:Double = -1.0
    
    when(oper){
        1 -> result = priNum + segNum
        2 -> result = priNum - segNum
        3 -> result = priNum * segNum
        4 -> result = priNum / segNum
        else -> return("Algo de errado não está certo!")
    }
    
    return("$result")
}
